常用sql语句

总结工作中常用的sql语句。

查看建表语句

show create table (dbname.)tablename

查看表结构语句

desc (dbname.)tablename或者show fields in (dbname.)tablename

查看库中表名

show tables (in dbname)

查看执行计划

explain (extended) select…

创建索引

ALTER TABLE table_name ADD INDEX index_name (column_list)
ALTER TABLE table_name ADD UNIQUE (column_list)
ALTER TABLE table_name ADD PRIMARY KEY (column_list)
CREATE INDEX index_name ON table_name (column_list)
CREATE UNIQUE INDEX index_name ON table_name (column_list)

查看索引

show index from tblname

插入

INSERT INTO sales_crm.team_area (team_id,parent_id,city,is_valid) VALUES (348,212,370300,1);

删除

DELETE FROM sales_crm.team_area WHERE id = 28;
DELETE FROM sales_crm.team_area WHERE id IN (28,29,30,31);

查询表中user_id字段不同order_type类型的数量

select sales_id,sales_name,order_type ,count(*) from sales_order group by sales_id,sales_name,order_type order by sales_id,sales_name,order_type;

查询表中user_id字段重复的数量

select user_id, count(1) as c from crm_client group by user_id having c > 1 order by c desc;

删除表中字段user_id重复数据，只留下其中一条

DELETE t1 FROM sale_cargo_info AS t1 LEFT JOIN (SELECT id,sale_id,cargo_active_count FROM sale_cargo_info
GROUP BY sale_id
HAVING COUNT(sale_id) >= 2) AS t2 ON t1.sale_id = t2.sale_id WHERE t1.id > t2.id;

表增加字段

ALTER TABLE infodb.bi_cargo_order_day ADD is_valid tinyint(1) NOT NULL DEFAULT 1 COMMENT ‘是否有效1-是，0-否’;
ALTER TABLE sales_crm.sales_visit_evaluate ADD (service_evaluate tinyint(1) NOT NULL DEFAULT ‘-1’ COMMENT ‘服务态度评价，0-不满意，1-满意，-1未评价’,function_evaluate tinyint(1) NOT NULL DEFAULT ‘-1’ COMMENT ‘产品功能评价，0-不满意，1-满意，-1未评价’,content varchar(256) NOT NULL DEFAULT ‘’ COMMENT ‘评价内容’);

修改表字段属性

ALTER TABLE notice MODIFY link_type INT(4) UNSIGNED NOT NULL DEFAULT ‘0’ COMMENT ‘链接类型’;

行子查询

SELECT FROM t1 WHERE ROW(1,2) = (SELECT column1, column2 FROM t2);
SELECT FROM article WHERE (title,content,uid) = (SELECT title,content,uid FROM blog WHERE bid=2);

嵌套查询

SELECT sb1,sb2,sb3 FROM (SELECT s1 AS sb1, s2 AS sb2, s3*2 AS sb3 FROM t1) AS sb WHERE sb1 > 1;
